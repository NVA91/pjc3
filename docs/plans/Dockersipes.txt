Erweiterte docker-compose.yml (mit cAdvisor, Node-Exporter, Loki)

text
version: "3.9"
services:
  # Bestehende Services + Monitoring Stack
  caddy:
    image: caddy:2-alpine
    ports: ["127.0.0.1:80:80", "127.0.0.1:443:443"]
    volumes:
      - caddy_data:/data
      - caddy_config:/config
      - ./Caddyfile:/etc/caddy/Caddyfile:ro
    networks: [home-net]

  # Container Monitoring
  cadvisor:
    image: gcr.io/cadvisor/cadvisor:latest
    restart: unless-stopped
    volumes:
      - /:/rootfs:ro
      - /var/run:/var/run:rw
      - /sys:/sys:ro
      - /var/lib/docker/:/var/lib/docker:ro
    networks: [home-net]

  node-exporter:
    image: prom/node-exporter:latest
    volumes:
      - /proc:/host/proc:ro
      - /sys:/host/sys:ro
      - /:/rootfs:ro
    command:
      - --path.procfs=/host/proc
      - --path.sysfs=/host/sys
    networks: [home-net]

  prometheus:
    image: prom/prometheus:latest
    volumes:
      - ./prometheus.yml:/etc/prometheus/prometheus.yml:ro
      - prometheus_data:/prometheus
    networks: [home-net]

  grafana:
    image: grafana/grafana:latest
    environment:
      - GF_SECURITY_ADMIN_PASSWORD=changeit123
    volumes: [grafana_data:/var/lib/grafana]
    networks: [home-net]

  # Loki (Logs)
  loki:
    image: grafana/loki:3.1.1
    restart: unless-stopped
    volumes:
      - ./loki-config.yml:/etc/loki/local-config.yaml
    command: -config.file=/etc/loki/local-config.yaml
    networks: [home-net]

  # Promtail (Log-Sammler)
  promtail:
    image: grafana/promtail:3.1.1
    volumes:
      - ./promtail-config.yml:/etc/promtail/config.yml
      - /var/lib/docker/containers:/var/lib/docker/containers:ro
    command: -config.file=/etc/promtail/config.yml
    networks: [home-net]
    depends_on: [loki]

networks: {home-net: {driver: bridge}}
volumes: [caddy_data, caddy_config, prometheus_data, grafana_data]

docker-compose.yml Template (eigene Projekte)

text
version: "3.9"
services:
  # Reverse Proxy (für alle *.local)
  caddy:
    image: caddy:2-alpine
    restart: unless-stopped
    ports:
      - "127.0.0.1:80:80"
      - "127.0.0.1:443:443"
    volumes:
      - caddy_data:/data
      - caddy_config:/config
      - ./Caddyfile:/etc/caddy/Caddyfile:ro
    networks: [home-net]

  # DEIN PROJEKT 1
  meinewebseite:
    build: ./projekte/webseite1    # Oder: image: dein/image
    restart: unless-stopped
    volumes:
      - ./projekte/webseite1/data:/app/data
    environment:
      - DB_HOST=mydb1
    networks: [home-net]
    depends_on:
      - mydb1

  mydb1:
    image: postgres:16-alpine
    environment:
      POSTGRES_PASSWORD: ${DB_PASS1}
    volumes: [webseite1_db:/var/lib/postgresql/data]
    networks: [home-net]

  # DEIN PROJEKT 2  
  meineapp:
    build: ./projekte/app2
    restart: unless-stopped
    networks: [home-net]
    depends_on: [app2_redis]

  app2_redis:
    image: redis:7-alpine
    networks: [home-net]

networks: {home-net: {driver: bridge}}
volumes: [caddy_data, caddy_config, webseite1_db]

Caddyfile (automatisch alle Services)

text
# Deine eigenen Projekte
meinewebseite.local {
  reverse_proxy meinewebseite:3000
  tls internal
}

meineapp.local {
  reverse_proxy meineapp:8080
}

# Wildcard für neue Projekte
*.local {
  reverse_proxy {upstream 80}
  tls internal
}

.env (Geheimnisse)

bash
DB_PASS1=dein_starkes_passwort123

Verwendung (5 Sekunden)

text
1. Neues Projekt? → Ordner `./projekte/neuername/` + Dockerfile
2. Service in docker-compose.yml hinzufügen
3. Zeile in Caddyfile: `neuername.local { reverse_proxy neuername:PORT }` 
4. docker compose up -d
5. https://neuername.local → Fertig!
